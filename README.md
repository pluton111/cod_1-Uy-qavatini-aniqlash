bu kodda hech qanday funkisya ishlatilmasdan, faqat arifmetik ammallar bajarilgan.
xonadon=int(input("Assalomu aleykum. \nHonadon raqamni kiriting:"))
w=xonadon/3
n=w%1
if n > 0:
    print(f"{xonadon}-honadon {int(w)+1}-qavatda joylashgan") 
else:
    print(f"{xonadon}-honadon {int(w)}-qavatda joylashgan")
